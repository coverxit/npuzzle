100  // The comparison function for the priority queue.
101  [&](const NPuzzleNode& a, const NPuzzleNode& b) {
102  // The element with less f(n) has higher priority,
103  // which actually constructs a min-heap.
104  // (The STL heap is a max-heap default)
105  int costA = GFunc(a) + hFunc(a);
106  int costB = GFunc(b) + hFunc(b);
107 
108  // If we got two nodes with the same priority,
109  // then we prefer the node with shallower depth.
110  if (costA == costB)
111  return GFunc(a) > GFunc(b);
112  else // Maintain a min-heap.
113  return costA > costB;
114  }
115  );
116 
117  visitedState[initialState] = true;
